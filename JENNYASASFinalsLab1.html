import { Component, signal } from '@angular/core';

import { CommonModule } from '@angular/common';

import { FormsModule } from '@angular/forms';

import { Firestore, collection, collectionData, addDoc, doc, deleteDoc, updateDoc } from

'@angular/fire/firestore';

import { Observable } from 'rxjs';

@Component({

selector: 'app-root',

standalone: true,

imports: [CommonModule, FormsModule],

templateUrl: './app.html',

styleUrls: ['./app.css']

})

export class App {

title = signal('APPDEV Angular with Firebase Demonstration with Student ManagementSystem');

name= signal('');

course = signal('');

userName = signal('');

password = signal ('');

email = signal('');

students: any[] = [];

constructor(private firestore: Firestore) {

const studentsCollection = collection(this.firestore, 'students');

collectionData(studentsCollection, { idField: 'id' })

.subscribe(data => {

this.students = data; // Assign to array so Angular detects changes

});

}

addStudent() {

const name = this.name();

const course = this.course();

const userName = this.userName();

const email = this.email();

const password = this.password();

if (name && course && userName && email && password ) {

const studentsCollection = collection(this.firestore, 'students');

addDoc(studentsCollection, { name, course, userName, email, password });

this.name.set('');

this.course.set('');

this.userName.set('');

this.email.set('');

this.password.set('');

}

}

deleteStudent(id: string) {

const studentDoc = doc(this.firestore, `students/${id}`);

deleteDoc(studentDoc);

}

updateStudent(id: string, newName: string, newCourse: String, newEmail: String, newUsername: String, newPassword: String ) {

const studentDoc = doc(this.firestore, `students/${id}`);

updateDoc(studentDoc, { name: newName, course: newCourse, email: newEmail, userName: newUsername, password:newPassword});

}

}

<div style="text-align:center; margin-top:20px;">

 <h1>{{ title() }}</h1>

  <input type="text" [value]="name()" (input)="name.set($any($event.target).value)" placeholder="Name">

 <input type="text" [value]="course()" (input)="course.set($any($event.target).value)" placeholder="Course">

 <input type="text" [value]="email()" (input)="email.set($any($event.target).value)" placeholder="Email">

 <input type="text" [value]="userName()" (input)="userName.set($any($event.target).value)" placeholder="Username">

 <input type="text" [value]="password()" (input)="password.set($any($event.target).value)" placeholder="Password">

 <button (click)="addStudent()">Add Student</button>

 <hr>

  <div *ngFor="let student of students" style="margin-bottom:10px;">

 <input [value]="student.name" (input)="student.name = $any($event.target).value">

 <input [value]="student.course" (input)="student.course = $any($event.target).value">

 <input [value]="student.email" (input)="student.email= $any($event.target).value">

 <input [value]="student.userName" (input)="student.userName = $any($event.target).value">

 <input [value]="student.password" (input)="student.password = $any($event.target).value">

 <button (click)="updateStudent(student.id, student.name,

 student.userName, student.course, student.email, student.password)">Update</button>

 <button (click)="deleteStudent(student.id)">Delete</button>

 </div>

 </div>

import { bootstrapApplication } from '@angular/platform-browser';

import { App } from './app/app';

import { provideFirebaseApp, initializeApp } from '@angular/fire/app';

import { provideFirestore, getFirestore } from '@angular/fire/firestore';

import { environment } from './environments/environment';

bootstrapApplication(App, {

providers: [

provideFirebaseApp(() => initializeApp(environment.firebase)),

provideFirestore(() => getFirestore())

]

}).catch(err => console.error(err));

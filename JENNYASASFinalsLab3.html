import { Component, OnInit } from '@angular/core';

import { CommonModule } from '@angular/common';

import { FormsModule } from '@angular/forms';

import { FirebaseService } from '../../services/firebase.service';

@Component({

 selector: 'app-evacuation-center',

 standalone: true,

 imports: [CommonModule, FormsModule],

 templateUrl: './evacuation-center.html',

 styleUrl: './evacuation-center.css'

})

export class EvacuationCenter implements OnInit {

 records: any[] = [];

 isEditing = false;

 editId: string | null = null;

  formData = {

   centerName: '',

   barangay: '',

   capacity: '',

   currentPopulation: '',

   status: ''

 };

 constructor(private firebaseService: FirebaseService) { }

 ngOnInit(): void {

   this.loadRecords();

 }

 loadRecords() {

   this.firebaseService.getCollection('evacuation_centers').subscribe(data => {

     this.records = data;

   });

 }

 handleSubmit() {

   if (!this.formData.centerName || !this.formData.barangay || !this.formData.capacity || !this.formData.currentPopulation || !this.formData.status) {

     alert('Please fill in all required fields');

     return;

   }

   if (this.isEditing && this.editId) {

     this.firebaseService.updateDocument('evacuation_centers', this.editId, this.formData)

       .then(() => {

         alert('Record updated successfully!');

         this.resetForm();

       });

   } else {

     this.firebaseService.addDocument('evacuation_centers', this.formData)

       .then(() => {

         alert('Record added successfully!');

         this.resetForm();

       });

   }

 }

 handleEdit(record: any) {

   this.formData = {

     centerName: record.centerName,

     barangay: record.barangay,

     capacity: record.capacity,

     currentPopulation: record.currentPopulation,

     status: record.status

   };

   this.isEditing = true;

   this.editId = record.id;

 }

 handleDelete(id: string) {

   if (confirm('Are you sure you want to delete this record?')) {

     this.firebaseService.deleteDocument('evacuation_centers', id)

       .then(() => alert('Record deleted successfully!'));

   }

 }

 resetForm() {

   this.formData = {

     centerName: '',

     barangay: '',

     capacity: '',

     currentPopulation: '',

     status: ''

   };

   this.isEditing = false;

   this.editId = null;

 }

}

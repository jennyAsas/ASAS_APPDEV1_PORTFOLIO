import { Component, signal } from '@angular/core';

import { CommonModule } from '@angular/common';

import { FormsModule } from '@angular/forms';

import { Firestore, collection, collectionData, addDoc, doc, deleteDoc, updateDoc } from

'@angular/fire/firestore';

import { Observable } from 'rxjs';

@Component({

selector: 'app-root',

standalone: true,

imports: [CommonModule, FormsModule],

templateUrl: './app.html',

styleUrls: ['./app.css']

})

export class DailyServiceRecord {

title = signal('APPDEV Angular with Firebase Demonstration with Student ManagementSystem');

name = signal('');

age = signal<number | null>(null);

date = signal('');

gender = signal('');

sicknessDiagnosed = signal('');

treatmentProvided = signal('');

attendingPhysician = signal('');

serviceRecords: any[] = [];

constructor(private firestore: Firestore) {

const serviceRecords= collection(this.firestore, 'Service Records');

collectionData(serviceRecords, { idField: 'id' })

.subscribe(data => {

this.serviceRecords = data; // Assign to array so Angular detects changes

});

}

addServiceRecords() {

const name = this.name();

const age = this.age();

const date =this.date();

const gender =this.gender();

const sicknessDiagnosed =this.sicknessDiagnosed();

const treatmentProvided =this.treatmentProvided();

const attendingPhysician =this.attendingPhysician();

if (name && age && date && gender && sicknessDiagnosed && treatmentProvided && attendingPhysician) {

const serviceRecords = collection(this.firestore, 'Service Records');

addDoc(serviceRecords, { name, age , date, gender, sicknessDiagnosed, treatmentProvided, attendingPhysician});

this.name.set('');

this.age.set(null);

this.gender.set('');

this.sicknessDiagnosed.set('');

this.treatmentProvided.set('');

this.attendingPhysician.set('');

}

}

deletePatient(id: string) {

const serviceRecords = doc(this.firestore, `students/${id}`);

deleteDoc(serviceRecords);

}

updatePatient(id: string, newName: string, newAge: number , newGender: String, newSicknessDiagnosed: String, newTreatmentProvided: String, newAttendingPhysician: String) {

const serviceRecords= doc(this.firestore, `students/${id}`);

updateDoc(serviceRecords, { name: newName, age: newAge, gender: newGender, sicknessDiagnosed: newSicknessDiagnosed, treatmentProvided: newTreatmentProvided, attendingPhysician: newAttendingPhysician});

}

}
